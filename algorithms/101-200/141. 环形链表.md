/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */

/**
 * @param {ListNode} head
 * @return {boolean}
 */
var hasCycle = function (head) {
    let dataMap = new Set()
    while (head) {
        if (dataMap.has(head)) {
            return true
        }
        dataMap.add(head)
        head = head.next
    }
    return false
};
