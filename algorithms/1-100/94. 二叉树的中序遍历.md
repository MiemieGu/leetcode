## 非递归思路


```
var inorderTraversal = function(root) {

    let a = []

    function tra(root){
        if (!root){
            return
        }
        if (root.left){
           tra(root.left)
        }
        a.push(root.val)
        if (root.right){
            tra(root.right)
        }
    }
    tra(root)
    return a
};
```
