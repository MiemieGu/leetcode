给定一个字符串 s 和一些长度相同的单词 words。找出 s 中恰好可以由 words 中所有单词串联形成的子串的起始位置。

注意子串要与 words 中的单词完全匹配，中间不能有其他字符，但不需要考虑 words 中单词串联的顺序。

```case
示例 1：
输入：
  s = "barfoothefoobarman",
  words = ["foo","bar"]
输出：[0,9]
解释：
从索引 0 和 9 开始的子串分别是 "barfoo" 和 "foobar" 。
输出的顺序不重要, [9,0] 也是有效答案。

示例 2：
输入：
  s = "wordgoodgoodgoodbestword",
  words = ["word","good","best","word"]
输出：[]
```

var findSubstring = function (s, words) {
    if (s == '' || words.length == 0) return []
    let dict = {},
        len = words[0].length * words.length
    let k = words[0].length
    for (let i = 0; i < words.length; i++) {
        if (dict[words[i]]) dict[words[i]]++
        else dict[words[i]] = 1
    }
    let ans = []
    let l = 0
    r = -1
    while (r < s.length) {
        r++
        while (r - l == len - 1) {
            let j = l
            let counter = {}
            console.log(j)
            while (j < r + 1) {
                let ss = s.slice(j, j + k)
                if (!dict.hasOwnProperty(ss)) break
                if (counter[ss]) counter[ss]++
                else counter[ss] = 1
                j += k
            }
            if (j - 1 == r) {
                if (same(counter, dict)) ans.push(l)
            }
            l++
        }
    }
    return ans
};

function same(m1, m2) {
    for (let k in m1) {
        if (m1[k] != m2[k]) return false
    }
    return true
}
